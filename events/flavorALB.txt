
namespace = flavor_alb

country_event = {
  id = flavor_alb.1
  title = flavor_alb.1.t
  desc = flavor_alb.1.d
  picture = COLONIZATION_eventPicture

  trigger = {
    OR = {
      tag = ALB
      was_tag = ALB
    }
    primary_culture = albanian
    any_owned_province = {
      OR = {
        culture = serbian
        culture = croatian
      }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 4
  }

  option = {#slavs
    name = flavor_alb.1.a
    ai_chance = { factor = 1 }
    trigger = {
      any_owned_province = {
				OR = {
					culture = serbian
					culture = croatian
				}
				NOT = { nationalism	= 1 }
				is_core = ROOT
      }
    }
    random_owned_province = {
      limit = {
				OR = {
					culture = serbian
					culture = croatian
				}
				NOT = { nationalism	= 1 }
				is_core = ROOT
      }
      change_culture = albanian
      change_religion = illyrian
    }
  }
}

namespace = flavor_slavs

country_event = {#uralic
  id = flavor_slavs.1
  title = flavor_slavs.1.t
  desc = flavor_slavs.1.d
  picture = COLONIZATION_eventPicture

  trigger = {
    culture_group = east_slavic
    any_owned_province = {
     has_owner_religion = yes
      OR = {
        culture = uralic
      }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 10
  }

  option = {
    name = flavor_slavs.1.a
    ai_chance = { factor = 1 }
    random_owned_province = {
      limit = {
       has_owner_religion = yes
       OR = {
         culture = uralic
       }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = russian
    }
  }
}


namespace = flavor_turkic

country_event = {
  id = flavor_turkic.1
  title = flavor_turkic.1.t
  desc = flavor_turkic.1.d
  picture = COLONIZATION_eventPicture

  trigger = {
    OR = {
      culture_group = Oghuz
    }
    any_owned_province = {
      culture_group = tartar
      NOT = {culture = bashkir }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 3
  }
  option = {#
    name = flavor_turkic.1.a
    ai_chance = { factor = 1 }
    trigger = {
      any_owned_province = {
        culture_group = tartar
        NOT = {culture = bashkir }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
    }
    random_owned_province = {
      limit = {
        culture_group = tartar
        NOT = {culture = bashkir }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = turkish
    }
  }
}

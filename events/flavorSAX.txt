#colonize polabia
country_event = {
  id = flavor_sax.16
  title = flavor_sax.16.t
  desc = flavor_sax.16.d
  picture = COLONIZATION_eventPicture

  trigger = {
    OR = {
      tag = SAX
      tag = THU
    }
    any_owned_province = {
      religion = germanic
      OR = {
        culture = polabian
        culture = czech
      }
      NOT = { culture_group = germanic }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 5
  }

  option = {#slavic
    name = flavor_sax.16.a
    ai_chance = { factor = 1 }
    trigger = {
      any_owned_province = {
        is_core = ROOT
        has_owner_religion = yes
        OR = {
          culture = polabian
          culture = czech
        }
        NOT = { nationalism	= 1 }
      }
    }
    random_owned_province = {
      limit = {
        has_owner_religion = yes
        OR = {
          culture = polabian
          culture = czech
        }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = east_saxon
    }
  }
}

#An Emisarry from the Senate
country_event = {
  id = flavor_ldn.2
  title = flavor_ldn.2.t
  desc = flavor_ldn.2.d
  picture = PROTESTANT_CATHOLIC_DEBATE_eventPicture
  fire_only_once = yes

  trigger = {
    exists = SPQ
    is_year = 1500
    total_development = 80
    OR = {
      tag = LDN
      tag = BRN
    }
  }

  option = {#Submit
    name = flavor_ldn.2.a
    ai_chance = { factor = 10 }
    add_prestige = -10
    add_legitimacy = 30
    add_stability = 2
    add_treasury = 500
    SPQ = {
      create_subject = {
        subject_type = roman_province_decen
        subject = ROOT
      }
    }
    add_country_modifier = {
      name = ldn_vassal_spq
      duration = -1
    }
  }
  option = {#Cooperate
    name = flavor_ldn.2.b
    ai_chance = { factor = 60 }
    add_legitimacy = 20
    add_dip_power = 100
    add_stability = 1
    create_alliance = SPQ
    SPQ = { add_historical_friend = ROOT }
    add_historical_friend = SPQ
    add_country_modifier = {
      name = ldn_coop_spq
      duration = -1
    }
  }
  option = {#Dismiss
    name = flavor_ldn.2.c
    ai_chance = { factor = 30 }
    add_legitimacy = -15
    add_prestige = 20
    add_mil_power = 100
    add_dip_power = -100
    add_stability = -1
    add_country_modifier = {
      name = ldn_rival_spq
      duration = -1
    }
    set_country_flag = roman_inheritor
    reverse_add_opinion = {
      who = SPQ
      modifier = rival
    }
    change_government = republic
    add_government_reform = roman_curia_reform
    set_government_rank = 3
    every_owned_province = {
      limit = {
        culture = britannian
      }
      change_culture = britannian_2
    }
    change_primary_culture = britannian_2
  }
}

#Cultural conversion when opposed to rome

country_event = {#Anglo-saxon
  id = flavor_ldn.3
  title = flavor_ldn.3.t
  desc = flavor_ldn.3.d
  picture = COLONIZATION_eventPicture

  trigger = {
    OR = {
      tag = BRN
      tag = LDN
    }
    has_country_modifier = ldn_rival_spq
    any_owned_province = {
      religion = deii
      OR = {
        culture = angle
        culture = seaxan
      }
      NOT = { culture_group = britannian_group }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 5
  }

  option = {#Anglo-saxon
    name = flavor_ldn.3.a
    ai_chance = { factor = 1 }
    trigger = {
      any_owned_province = {
        is_core = ROOT
        religion = deii
        OR = {
          culture = angle
          culture = seaxan
        }
        NOT = { nationalism	= 1 }
      }
    }
    random_owned_province = {
      limit = {
        religion = deii
        OR = {
          culture = angle
          culture = seaxan
        }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = flavian
    }
  }
}

	major = yes


	option = {		#cast out of the Empire

		}
		BUL = {
			set_in_empire = no
			add_country_modifier = {
				name = cast_out_bul
				duration = 7300
			}
		}
	}
	option = {	#punish bulgaria
		name = "flavor_bul.2.b"
		ai_chance = { factor = 30 }
		every_country = {
			limit = {
				is_part_of_hre = yes
			}
			add_opinion = {
			    who = BUL
			    modifier = barbarian_bul
			}
			BUL = {country_event = { id = flavor_bul.4 }}
		}
	}

	option = { #this is just a raid in a war, it's quite common
		name = "flavor_bul.2.c"
		ai_chance = { factor = 20 }
		add_imperial_influence = -30
	}
}

	major = yes


	option = {#accept
		name = flavor_bul.4.a
		ai_chance = { factor = 10 }
		BUL = {
			add_treasury = -1000
		}
		every_country = {
			limit = {
				NOT = {
					border_distance = {
						who = BUL
						distance = 1
					}
				}
				is_part_of_hre = yes
			}
			add_treasury = 200
		}
	}

	option = { #refuse
    name = flavor_bul.4.b

		}
		set_in_empire = no
		add_country_modifier = {
			name = cast_out_bul
			duration = 7300
		}
  }
}

country_event = {#jewish issue
  id = flavor_kip.2
  title = flavor_kip.2.t
  desc = flavor_kip.2.d
  picture = COLONIZATION_eventPicture

  is_triggered_only = yes

  option = {#tolerate
    name = flavor_kip.2.a
    ai_chance = { factor = 1 }
		set_country_flag = tolerate_jews_f
		every_owned_province = {
			limit = {
				religion = jewish
			}
			add_province_modifier = {
				name = tolerate_jews
				duration = -1
			}
		}
  }
	option = {#not tolerate
    name = flavor_kip.2.b
    ai_chance = { factor = 1 }
		add_country_modifier = {
			name = expulse_jews
			duration = 7300
		}
  }
	option = {#convert
    name = flavor_kip.2.c
    ai_chance = { factor = 1 }
		change_religion = jewish
  }
}

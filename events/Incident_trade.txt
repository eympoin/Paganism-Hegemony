	id = new_trade_roads.3
	title = new_trade_roads.3.t
	desc = new_trade_roads.3.d
	picture = HRE_eventPicture

	fire_only_once = yes
	trigger = {
		is_emperor = yes
		has_global_flag = the_emperor_go_to_india
		any_province = {
			area = basra_area
			owner = {
				war_with = emperor
			}
		}
		war_score = 50
	}
	mean_time_to_happen = {
		months = 1
	}

	option = {
		name = new_trade_roads.3.a
		ai_chance = {
			factor = 1
		}
		every_province ={
			limit = {
				area = basra_area
				owner = {
					war_with = emperor
				}
			}
			set_in_empire = yes
		}


		every_country = {
			limit = {
				war_with = emperor
				OR = {
					country_or_non_sovereign_subject_holds  = 2311
					country_or_non_sovereign_subject_holds  = 2312
					country_or_non_sovereign_subject_holds  = 4288
					country_or_non_sovereign_subject_holds  = 408
				}
			}
		white_peace = emperor
		}
		MES = {
			change_religion = theoi
		}
		every_province ={
			limit = {
				area = basra_area
				is_part_of_hre = yes
			}
			cede_province	= MES
			add_core = MES
		}

		408 = {
			add_trade_modifier = {
				who = emperor
				duration = -1
				power = 60
				key = spice_road
				}
		}
		set_global_flag = control_of_mesopotamia

			limit = {

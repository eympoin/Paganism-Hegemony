country_event = {#Welsh/cornish/breton
  id = flavor_wrm.1
  title = flavor_wrm.1.t
  desc = flavor_wrm.1.d
  picture = COLONIZATION_eventPicture

  trigger = {
    OR = {
      tag = WRM
      was_tag = WRM
    }
    culture_group = neo-roman
    any_owned_province = {
     has_owner_religion = yes
      OR = {
        culture = welsh
        culture = cornish
        culture = breton
      }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 5
  }

  option = {
    name = flavor_wrm.1.a
    ai_chance = { factor = 1 }
    random_owned_province = {
      limit = {
       has_owner_religion = yes
        OR = {
          culture = welsh
          culture = cornish
          culture = breton
        }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = britannian
    }
  }
}

country_event = {#russian
  id = flavor_tur.3
  title = flavor_tur.3.t
  desc = flavor_tur.3.d
  picture = COLONIZATION_eventPicture

  trigger = {
    has_country_flag = gogo_turks
    primary_culture = turkish
    any_owned_province = {
      OR = {
        AND = {
          owner = {has_matching_religion = slavic}
          religion = slavic
        }
        has_owner_religion = yes
      }

       OR = {
         culture = ruthenian
         culture = byelorussian
         culture = ryazanian
         culture = russian
         culture = novgorodian
       }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 4
  }

  option = {
    name = flavor_tur.3.a
    ai_chance = { factor = 1 }
    random_owned_province = {
      limit = {
      OR = {
        AND = {
          owner = {has_matching_religion = slavic}
          religion = slavic
        }
        has_owner_religion = yes
      }

       OR = {
         culture = ruthenian
         culture = byelorussian
         culture = ryazanian
         culture = russian
         culture = novgorodian
       }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = tatar_turk
    }
  }
}

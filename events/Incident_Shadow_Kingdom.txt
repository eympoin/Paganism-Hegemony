	major = yes

	trigger = {
		is_emperor = yes
		has_global_flag = emperor_trying_to_avert_shadow_kingdom

			}

		}

				is_part_of_hre = yes
				capital_scope = { region = egypt_region }
				will_leave_for_shadow_kingdom_trigger = yes

					is_subject = no
					AND = {
						is_subject = yes
						overlord = { # overlord chooses for their vassals if eligible to leave themselves
							NOT = {
								is_part_of_hre = yes
								capital_scope = { region = egypt_region }

					}

			}
			custom_tooltip = shadow_kingdom_will_leave_info_tt
		}
		every_country = {
			limit = {
				is_part_of_hre = yes
				capital_scope = { region = egypt_region }
				will_leave_for_shadow_kingdom_trigger = no
				OR = {
					is_subject = no
					AND = {
						is_subject = yes
						overlord = { # overlord chooses for their vassals if eligible to leave themselves
							NOT = {
								is_part_of_hre = yes
								capital_scope = { region = egypt_region }
							}
						}
					}
				}
			}
			custom_tooltip = shadow_kingdom_will_stay_info_tt
		}
		custom_tooltip = shadow_kingdom_result_ia_tooltip
		every_province = {
			limit = {
				is_part_of_hre = yes
				region = egypt_region
				owner = {
					is_part_of_hre = no
				}
			}
			set_in_empire = no
		}
	}
	option = {

	}

# Italian country leaves or stays
country_event = {
	id = shadow_kingdom.4
	title = hre_event.1.t
	desc = hre_event.1.d
	picture = HRE_eventPicture

	is_triggered_only = yes

	option = { # remain
		ai_chance = {
			factor = 0
			modifier = {
				factor = 1
				will_leave_for_shadow_kingdom_trigger = no
			}
		}
		name = hre_event.1.b
		if = {
			limit = {
				has_global_flag = emperor_trying_to_avert_shadow_kingdom
			}
			emperor = {
				add_imperial_influence = 5
			}
		}
		add_prestige = -20
		custom_tooltip = shadow_kingdom_triggered_modifier_tt
	}
	option = { # leave
		ai_chance = {
			factor = 1000
			modifier = {
				factor = 1000
				will_leave_for_shadow_kingdom_trigger = yes
			}
		}
		name = hre_event.1.a
		set_in_empire = no
		set_country_flag = permanently_left_hre_flag
		emperor = {
			add_opinion = {
				who = ROOT
				modifier = opinion_left_hre
			}
		}
		if = {
			limit = {
				any_country = {
					is_subject_of = ROOT
					capital_scope = {
						region = egypt_region
					}
					is_part_of_hre = yes
				}
			}
			every_subject_country = {
				limit = {
					capital_scope = {
						region = egypt_region
					}
					is_part_of_hre = yes
				}
				set_in_empire = no
			}
		}
		if = {
			limit = {
				has_global_flag = emperor_trying_to_avert_shadow_kingdom
			}
			emperor = {
				add_imperial_influence = -10
			}
		}
	}
}

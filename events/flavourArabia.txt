country_event = {# Egyptian/Ptolemaic/cyrenian
  id = flavor_mda.6
  title = flavor_mda.6.t
  desc = flavor_mda.6.d
  picture = COLONIZATION_eventPicture

  trigger = {
    OR = {
      culture_group = turko_semitic
      religion_group = muslim
    }
    any_owned_province = {
      religion_group = muslim
      OR = {
        culture = al_misr_arabic
        culture = ptolemaic
        culture = cyrenian
      }
      NOT = { culture_group = turko_semitic }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 2
  }

  option = {
    name = flavor_mda.6.b
    ai_chance = { factor = 1 }
    trigger = {
      any_owned_province = {
        is_core = ROOT
        religion_group = muslim
        OR = {
        culture = al_misr_arabic
        culture = ptolemaic
        culture = cyrenian
        }
        NOT = { nationalism	= 1 }
      }
    }
    random_owned_province = {
      limit = {
        religion_group = muslim
          OR = {
          culture = al_misr_arabic
          culture = ptolemaic
          culture = cyrenian
          }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = Egyptian_Arabic
    }
  }
}

country_event = {#gaul
  id = flavor_ire.1
  title = flavor_ire.1.t
  desc = flavor_ire.1.d
  picture = COLONIZATION_eventPicture

  trigger = {
    OR = {
      AND = {
        culture_group = gaelic
        NOT = {culture = irish}
      }
      AND = {
        culture_group = gaelic
        has_country_flag = irish_unlock_conversion_gallia
      }
    }
    any_owned_province = {
     has_owner_religion = yes
      OR = {
        culture = salian
        culture = rhenan
        culture = lingerian
        culture = aquitain
        culture = narbonnais
        culture = lugdunian
        culture = franko-roman
      }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 3
  }

  option = {
    name = flavor_ire.1.a
    ai_chance = { factor = 1 }
    random_owned_province = {
      limit = {
       has_owner_religion = yes
       OR = {
       culture = salian
       culture = rhenan
       culture = lingerian
       culture = aquitain
       culture = narbonnais
       culture = lugdunian
       culture = franko-roman
       }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = gallian
    }
  }
}

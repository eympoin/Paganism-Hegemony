namespace = flavor_ire


country_event = {#roman
  id = flavor_ire.1
  title = flavor_ire.1.t
  desc = flavor_ire.1.d
  picture = COLONIZATION_eventPicture

  trigger = {
    culture_group = gaelic
    any_owned_province = {
     has_owner_religion = yes
      OR = {
        culture = aquitain
				culture = narbonnais
      }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 5
  }

  option = {
    name = flavor_ire.2.a
    ai_chance = { factor = 1 }
    random_owned_province = {
      limit = {
       has_owner_religion = yes
       OR = {
				 culture = aquitain
				 culture = narbonnais
				 culture = lugdunian
				 culture = franko-roman
       }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = gallian
    }
  }
}

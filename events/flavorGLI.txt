# A plot for two thrones
namespace = flavor_nar
country_event = {
	id = flavor_nar.3
	title = "flavor_nar.EVTNAME3"
	desc = "flavor_nar.EVTDESC3"
	picture = COURT_eventPicture
	major = yes

	fire_only_once = yes


	trigger = {
		OR{
			AND{
				tag = NAR
				has_ruler = "Valerian"
				AQI = {
					exists = yes
					has_ruler = "Claudius"
					ai = yes
					was_player = no
					}

				}
			AND{
				tag = AQI
				ai = no
				has_ruler = "Claudius"
				NAR = {
					exists = yes
					has_ruler = "Valerian"
					ai = yes
					was_player = no
					}
				}
			}
		}
	mean_time_to_happen = {
		months = 50
	}

	option = {		# unite
		name = "flavor_nar.EVTOPTA3"
		ai_chance = { factor = 75 }
		if = {
			limit = { tag = NAR }
			create_union = AQI
			remove_historical_friend = AQI
			AQI = {
				remove_historical_friend = NAR
			}
		}
		if = {
			limit = { tag = AQI }
			create_union = NAR
			remove_historical_friend = NAR
			NAR = {
				remove_historical_friend = AQI
			}
		}
		random_owned_province = {
			limit = {
				is_capital = no
			}
			spawn_rebels = {
				type = noble_rebels
				size = 2
			}
		}
		add_stability = -1
	}
	option = {		# do not unite
		name = "flavor_nar.EVTOPTB3"
		ai_chance = { factor = 25 }
		add_stability_or_adm_power = yes
		add_prestige = 20

	}
}


country_event = {#franks
  id = flavor_nar.1
  title = flavor_nar.1.t
  desc = flavor_nar.1.d
  picture = COLONIZATION_eventPicture

  trigger = {
		OR = {
			tag = NAR
			tag = AQI
			tag = GLI
		}
    culture_group = neo-roman
    any_owned_province = {
			has_owner_religion = yes
      OR = {
        culture = lingerian
        culture = salian
        culture = rhenan
				culture = occitan
      }
      NOT = { culture_group = britannian_group }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 5
  }

  option = {
    name = flavor_nar.1.a
    ai_chance = { factor = 1 }
    random_owned_province = {
      limit = {
				has_owner_religion = yes
        OR = {
          culture = lingerian
          culture = salian
          culture = rhenan
					culture = occitan
        }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = franko-roman
    }
  }
}

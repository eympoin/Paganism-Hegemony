country_event = {#roman conversion
  id = flavor_athens.2
  title = flavor_athens.2.t
  desc = flavor_athens.2.d
  picture = COLONIZATION_eventPicture

  trigger = {
    culture_group = neo_roman
    any_country = {has_country_flag = greek_influence_in_the_west}
    any_owned_province = {
      is_capital = no
      OR = {
        religion = deii
        religion = theoi
        religion = mithraic
      }
      OR = {
        culture_group = neo_roman
      }
      NOT = { nationalism	= 1 }
      is_core = ROOT
    }
  }

  mean_time_to_happen = {
    years = 15
  }

  option = {
    name = flavor_athens.2.a
    ai_chance = { factor = 1 }
    random_owned_province = {
      limit = {
       has_owner_religion = yes
        OR = {
          culture_group = neo_roman
        }
        NOT = { nationalism	= 1 }
        is_core = ROOT
      }
      change_culture = greek
    }
  }
  option = {
    name = flavor_athens.2.b
    ai_chance = { factor = 1 }
    add_dip_power = -75
  }
}

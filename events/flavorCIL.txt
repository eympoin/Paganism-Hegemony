
namespace = flavor_cil

# birth of cilician_piracy
country_event = {
	id = flavor_cil.1
	title = "flavor_cil.1.t"
	desc = "flavor_cil.1.d"
	picture = BARBARY_PIRATES_eventPicture
	fire_only_once = yes
	trigger = {
		tag = CIL
		is_year = 1479
	}


	option = {		# embrace piracy
		name = "flavor_cil.1.a"
		ai_chance = { factor = 100 }
		add_country_modifier = {
			name = cilician_pirates
			duration = -1
		}
		set_global_flag = cilicia_piracy
		every_province = {
			limit = {
				OR = {
					region = syria_region
					area = caria_area
				}
				has_port = yes
			}
			add_province_modifier = {
				name = cilician_piracy_mod
				duration = -1
			}
		}
		every_country = {
			limit = {
				is_part_of_hre = yes
			}
			country_event = {id = flavor_cil.3 days = 15}
		}
	}

	option = {		# do nothing
		name = "flavor_cil.1.b"
		ai_chance = { factor = 0 }
		every_province = {
			limit = {
				OR = {
					region = syria_region
					area = caria_area
					area = cukurova_area
				}
				has_port = yes
			}
			add_province_modifier = {
				name = cilician_piracy_mod
				duration = 7300
			}
		}
		set_global_flag = end_cilician_piracy
	}


	option = {		# fight piracy
		name = "flavor_cil.1.c"
		ai_chance = { factor = 0 }
		set_global_flag = end_cilician_piracy
		every_province = {
			limit = {
				OR = {
					region = syria_region
					area = caria_area
					area = cukurova_area
				}
				has_port = yes
			}
			add_province_modifier = {
				name = cilician_piracy_mod_less
				duration = 7300
			}
		}
	}
}
#end piracy

			limit = {

}
